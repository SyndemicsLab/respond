53Eigen::VectorXd Behavior(Eigen::VectorXd &state,
54 const std::vector<Eigen::MatrixXd> &transition) {
55 if (transition.size() != 1) {
56 throw std::runtime_error(
57 "Behavior Transitions must have 1 Transition Matrix.");
58 }
59 if (state.rows() != transition[0].cols()) {
60 std::stringstream ss;
61 ss << "Unable to multiply behavior transition with "
62 "state, mismatched sizes. State size is (";
63 ss << state.rows() << ", " << state.cols();
64 ss << ") and transition size is (" << transition[0].rows() << ", ";
65 ss << transition[0].cols() << ").";
66 throw std::runtime_error(ss.str());
67 }
68 state = transition[0] * state;
69 return state;
70}

103Eigen::VectorXd Overdose(Eigen::VectorXd &state,
104 const std::vector<Eigen::MatrixXd> &transition) {
105 if (transition.size() != 2) {
106 throw std::runtime_error(
107 "Overdose Transitions must have 2 Transition Matrices.");
108 }
109
110 if (state.size() != transition[0].size()) {
111 throw std::runtime_error("Overdose Vector is not the same "
112 "size as the state vector.");
113 }
114
115 Eigen::VectorXd overdoses = state.cwiseProduct(transition[0]); // overdose
116
117 if (overdoses.size() != transition[1].size()) {
118 throw std::runtime_error("Fatal Overdose Vector is not the same "
119 "size as the state vector.");
120 }
121
122 auto fods = overdoses.cwiseProduct(transition[1]); // negatives
123
124 Eigen::VectorXd ods_moves = state - fods; // put the fods back
125 ods_moves.bottomRows(state.size() / 3) =
126 state.bottomRows(state.size() / 3) +
127 fods(Eigen::seqN(state.size() / 3, state.size() / 3));
128
129 state = ods_moves; // transition fatal overdoses
130 return overdoses; // return number of total overdoses
131}
