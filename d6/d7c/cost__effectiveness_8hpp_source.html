31inline Eigen::VectorXd Discount(const Eigen::VectorXd &data,
32 double discount_rate, int week,
33 bool is_discrete = true,
34 double total_weeks = 52.0) {
35 double discount =
36 (is_discrete) ? (1 / pow((1.0 + (discount_rate) / total_weeks), week))
37 : (exp(-discount_rate * (week / total_weeks)));
38 return data - Eigen::VectorXd::Constant(data.size(), discount);
39}

73inline CostStamp StampCosts(const Eigen::VectorXd &state,
74 const Eigen::VectorXd &healthcare_costs,
75 const Eigen::VectorXd &aod_costs,
76 const Eigen::VectorXd &fod_costs,
77 const Eigen::VectorXd &pharma_costs,
78 const Eigen::VectorXd &treatment_costs) {
79 if (!CheckVectorLengths(state, healthcare_costs, aod_costs, fod_costs,
80 pharma_costs, treatment_costs)) {
81 // log error: state and cost vectors size mismatch
82 return CostStamp{};
83 }
84 CostStamp cost_stamp;
85 cost_stamp.healthcare = state.cwiseProduct(healthcare_costs);
86 cost_stamp.non_fatal_overdoses = state.cwiseProduct(aod_costs);
87 cost_stamp.fatal_overdoses = state.cwiseProduct(fod_costs);
88 cost_stamp.pharmaceuticals = state.cwiseProduct(pharma_costs);
89 cost_stamp.treatments = state.cwiseProduct(treatment_costs);
90 return cost_stamp;
91}

97inline Eigen::VectorXd StampUtilities(const Eigen::VectorXd &state,
98 const Eigen::VectorXd &utility) {
99 if (state.size() != utility.size()) {
100 // log error: state and utility size mismatch
101 return Eigen::VectorXd::Ones(state.size());
102 }
103 return state.cwiseProduct(utility);
104}

117inline CostsOverTime StampCostsOverTime(
118 const HistoryOverTime &history_over_time,
119 const Eigen::VectorXd &healthcare_costs, const Eigen::VectorXd &aod_costs,
120 const Eigen::VectorXd &fod_costs, const Eigen::VectorXd &pharma_costs,
121 const Eigen::VectorXd &treatment_costs, bool discount = false,
122 double discount_rate = 0.0, double total_weeks = 52.0) {
123 CostsOverTime costs_over_time;
124 for (const auto &kv : history_over_time) {
125 costs_over_time[kv.first] =
126 StampCosts(kv.second.state, healthcare_costs, aod_costs, fod_costs,
127 pharma_costs, treatment_costs);
128 if (discount) {
129 DiscountCostStamp(costs_over_time[kv.first], discount_rate,
130 kv.first, true, total_weeks);
131 }
132 }
133 return costs_over_time;
134}

144inline Eigen::VectorXd StampUtilitiesOverTime(const HistoryOverTime &history,
145 const Eigen::VectorXd &utility,
146 UtilityType util_type,
147 bool discount = false,
148 double discount_rate = 0.0,
149 double total_weeks = 52.0) {
150 Eigen::VectorXd utilities;
151 bool is_first_iteration = true;
152 for (const auto &kv : history) {
153 auto stamp = StampUtilities(kv.second.state, utility);
154 if (discount) {
155 stamp = Discount(stamp, discount_rate, kv.first);
156 }
157 // If this is the first entry in the map, set the utilities vector
158 if (is_first_iteration) {
159 utilities = stamp;
160 is_first_iteration = false;
161 }
162
163 switch (util_type) {
164 case UtilityType::kMin:
165 utilities = utilities.cwiseMin(stamp);
166 break;
167 case UtilityType::kMult:
168 utilities = utilities.cwiseProduct(stamp);
169 break;
170 default:
171 break;
172 }
173 }
174 return utilities;
175}

189inline CostPerspectives
190CalculatePerspectives(const HistoryOverTime &history_over_time,
191 const std::vector<std::string> &perspectives,
192 const std::vector<Eigen::VectorXd> &healthcare_costs,
193 const std::vector<Eigen::VectorXd> &aod_costs,
194 const std::vector<Eigen::VectorXd> &fod_costs,
195 const std::vector<Eigen::VectorXd> &pharma_costs,
196 const std::vector<Eigen::VectorXd> &treatment_costs,
197 bool discount = false, double discount_rate = 0.0,
198 double total_weeks = 52.0) {
199 CheckVectorLengths(perspectives, healthcare_costs, aod_costs, fod_costs,
200 pharma_costs, treatment_costs);
201 CostPerspectives cost_perspectives;
202 for (int i = 0; i < perspectives.size(); ++i) {
203 cost_perspectives[perspectives[i]] = StampCostsOverTime(
204 history_over_time, healthcare_costs[i], aod_costs[i], fod_costs[i],
205 pharma_costs[i], treatment_costs[i], discount, discount_rate,
206 total_weeks);
207 }
208 return cost_perspectives;
209}

216inline double CalculateLifeYears(const HistoryOverTime &history,
217 bool discount = false,
218 double discount_rate = 0.0,
219 double total_weeks = 52.0) {
220 if (history.empty()) {
221 // log no state vector
222 return 0.0;
223 }
224
225 if (history.size() != total_weeks) {
226 // warn that history does not have 52 weeks
227 return 0.0;
228 }
229
230 Eigen::VectorXd running_total =
231 Eigen::VectorXd::Zero(history.begin()->second.state.size());
232
233 for (int t = 0; t < history.size(); ++t) {
234 auto state = history.at(t).state;
235 running_total +=
236 ((discount) ? Discount(state, discount_rate, t, true, total_weeks)
237 : state);
238 }
239 auto result = running_total.sum();
240
241 // dividing by 52 to switch from life weeks to life years
242 return result / 52.0;
243}

248inline std::vector<double> CalculateTotalCosts(const CostsOverTime &costs) {
249 std::vector<double> result;
250 for (const auto &cost : costs) {
251 result.push_back(cost.second.healthcare.sum() +
252 cost.second.non_fatal_overdoses.sum() +
253 cost.second.fatal_overdoses.sum() +
254 cost.second.pharmaceuticals.sum() +
255 cost.second.treatments.sum());
256 }
257 return result;
258}
