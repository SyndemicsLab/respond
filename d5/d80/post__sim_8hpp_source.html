30inline data_ops::Matrix3d CalculateDiscount(const data_ops::Matrix3d &data,
31 double discountRate, int N,
32 bool isDiscrete = true) {
33 double discountConstant = (isDiscrete)
34 ? (1 / pow((1.0 + (discountRate) / 52.0), N))
35 : (exp(-discountRate * (N / 52)));
36
37 data_ops::Matrix3d discount =
38 data_ops::CreateMatrix3d(data.dimension(0), data.dimension(1),
39 data.dimension(2))
40 .setConstant(discountConstant);
41 data_ops::Matrix3d result = data - discount;
42 return result;
43}

52inline data_ops::CostList
53CalculateCosts(const data_ops::History &history,
54 const data_ops::CostLoader &cost_loader,
55 const std::vector<std::string> &perspectives,
56 bool discount = false, double discount_rate = 0.0) {
57 data_ops::CostList costs;
58
59 for (std::string perspective : perspectives) {
60 data_ops::Cost cost;
61 cost.perspective = perspective;
62 cost.healthcare_cost = data_ops::MultiplyTimedMatrix3dByMatrix(
63 history.state_history,
64 cost_loader.GetHealthcareUtilizationCost(perspective));
65 cost.pharma_cost = data_ops::MultiplyTimedMatrix3dByMatrix(
66 history.state_history,
67 cost_loader.GetPharmaceuticalCost(perspective));
68 cost.treatment_cost = data_ops::MultiplyTimedMatrix3dByMatrix(
69 history.state_history,
70 cost_loader.GetTreatmentUtilizationCost(perspective));
71 cost.non_fatal_overdose_cost = data_ops::MultiplyTimedMatrix3dByDouble(
72 history.overdose_history,
73 cost_loader.GetNonFatalOverdoseCost(perspective));
74 cost.fatal_overdose_cost = data_ops::MultiplyTimedMatrix3dByDouble(
75 history.overdose_history,
76 cost_loader.GetFatalOverdoseCost(perspective));
77
78 if (discount) {
79 for (int i = 0; i < history.state_history.size(); ++i) {
80 cost.healthcare_cost[i] = CalculateDiscount(
81 cost.healthcare_cost[i], discount_rate, i);
82 cost.pharma_cost[i] =
83 CalculateDiscount(cost.pharma_cost[i], discount_rate, i);
84 cost.treatment_cost[i] =
85 CalculateDiscount(cost.treatment_cost[i], discount_rate, i);
86 cost.non_fatal_overdose_cost[i] = CalculateDiscount(
87 cost.non_fatal_overdose_cost[i], discount_rate, i);
88 cost.fatal_overdose_cost[i] = CalculateDiscount(
89 cost.fatal_overdose_cost[i], discount_rate, i);
90 }
91 }
92 costs.push_back(cost);
93 }
94 return costs;
95}

104inline data_ops::TimedMatrix3d
105CalculateUtilities(const data_ops::History &history,
106 const data_ops::UtilityLoader &utility_loader,
107 data_ops::UtilityType util_type, bool discount = false,
108 double discount_rate = 0.0) {
109 std::vector<data_ops::Matrix3d> utilityMatrices = {
110 utility_loader.GetBackgroundUtility("utility"),
111 utility_loader.GetOUDUtility("utility"),
112 utility_loader.GetSettingUtility("utility")};
113
114 data_ops::Matrix3d util;
115 switch (util_type) {
116 case data_ops::UtilityType::kMin:
117 util = data_ops::Matrix3dVectorMinimum(utilityMatrices);
118 break;
119 case data_ops::UtilityType::kMult:
120 util = data_ops::Matrix3dVectorMultiplied(utilityMatrices);
121 break;
122 default:
123 break;
124 }
125
126 data_ops::TimedMatrix3d utilities =
127 data_ops::MultiplyTimedMatrix3dByMatrix(history.state_history, util);
128
129 if (!discount) {
130 return utilities;
131 }
132
133 for (auto kv : utilities) {
134 kv.second = CalculateDiscount(kv.second, discount_rate, kv.first);
135 }
136 return utilities;
137}

144inline double CalculateLifeYears(const data_ops::History &history,
145 bool provideDiscount = false,
146 double discountRate = 0.0) {
147 if (history.state_history.size() <= 0) {
148 // log no state vector
149 return 0.0;
150 }
151
152 data_ops::Matrix3d running_total(
153 history.state_history.begin()->second.dimensions());
154 running_total = running_total.setZero();
155
156 for (int t = 0; t < history.state_history.size(); ++t) {
157 running_total +=
158 ((provideDiscount) ? CalculateDiscount(history.state_history.at(t),
159 discountRate, t)
160 : history.state_history.at(t));
161 }
162 Eigen::Tensor<double, 0> result = running_total.sum();
163
164 // dividing by 52 to switch from life weeks to life years
165 return result(0) / 52.0;
166}

171inline std::vector<double>
172CalculateTotalCosts(const data_ops::CostList &cost_list) {
173 std::vector<double> result;
174 for (data_ops::Cost cost : cost_list) {
175 double totalCost =
176 data_ops::TimedMatrix3dSummedOverDimensions(cost.healthcare_cost) +
177 data_ops::TimedMatrix3dSummedOverDimensions(
178 cost.fatal_overdose_cost) +
179 data_ops::TimedMatrix3dSummedOverDimensions(
180 cost.non_fatal_overdose_cost) +
181 data_ops::TimedMatrix3dSummedOverDimensions(cost.pharma_cost) +
182 data_ops::TimedMatrix3dSummedOverDimensions(cost.treatment_cost);
183 result.push_back(totalCost);
184 }
185 return result;
186}
