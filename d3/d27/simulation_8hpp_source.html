42 void AddModel(const std::unique_ptr<Model> &model) {

53 std::vector<std::string> ret;

60 void ClearModels() { _models.clear(); }

62 const std::vector<std::map<std::string, std::vector<Eigen::VectorXd>>>
63 GetModelHistories() const {
64 std::vector<std::map<std::string, std::vector<Eigen::VectorXd>>> ret;
65 int model_idx = 0;
66 for (const auto &model : _models) {
67 std::map<std::string, std::vector<Eigen::VectorXd>> inner_ret;
68 for (const auto &kv : model->GetHistories()) {
69 inner_ret[kv.first] = kv.second.GetStateAsVector();
70 }
71 ret.push_back(inner_ret);
72 model_idx++;
73 }
74 return ret;
75 }
76
77 const std::vector<std::pair<std::string, std::string>>
78 GetModelHistoryNames() const {
79 std::vector<std::pair<std::string, std::string>> ret;
80 for (const auto &model : _models) {
81 for (const auto &kv : model->GetHistories()) {
82 std::pair<std::string, std::string> p = {model->GetModelName(),
83 kv.first};
84 ret.push_back(p);
85 }
86 }
87 return ret;
88 }
89
90 std::string GetLogName() const { return _log_name; }
91
92 // Copying object
93 Simulation(const Simulation &other) : _log_name(other.GetLogName()) {
94 ClearModels();
95 for (const auto &m : other.GetModels()) {
96 _models.push_back(m->clone());
97 }
98 }
99 Simulation &operator=(const Simulation &other) {
100 if (this != &other) {
101 ClearModels();
102 _log_name = other.GetLogName();
103 for (const auto &m : other.GetModels()) {
104 _models.push_back(m->clone());
105 }
106 }
107 return *this;
108 }
109
110private:
111 std::string _log_name;
112 std::vector<std::unique_ptr<Model>> _models;
113};
