27class Transition {
28public:
29 virtual ~Transition() = default;
30 // Run the execute function and return the final state. Do not edit the
31 // parameter state, but do edit the history provided. Nothing in the
32 // Transition object should change.
33 virtual Eigen::VectorXd
34 Execute(const Eigen::VectorXd &s,
35 std::map<std::string, History> &h) const = 0;
36 // Add a Transition Matrix to the set. We have no need to edit it once it's
37 // been added, just use it. Thus, we don't need full ownership (reference)
38 // and can accept the const type.
39 virtual void AddTransitionMatrix(const Eigen::MatrixXd &m) = 0;
40 // Get the name of the Transition. No need to edit the object and do not
41 // need user to edit the name.
42 virtual std::string GetTransitionName() const = 0;
43 // Clear out all the stored Eigen::MatrixXd values
44 virtual void ClearTransitionMatrices() = 0;
45 // Get Log Name
46 virtual std::string GetLogName() const = 0;
47 // Clone
48 Transition(const Transition &) = delete;
49 Transition &operator=(const Transition &) = delete;
50 virtual std::unique_ptr<Transition> clone() const = 0;
51
52protected:
53 // default constructor required for subclasses, but do not want people to
54 // use this
55 Transition() = default;
56};
