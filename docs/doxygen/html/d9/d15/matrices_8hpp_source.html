32inline Matrix3d CreateMatrix3d(const int x_1, const int x_2, const int x_3) {

42inline void PrintMatrix3d(const Matrix3d &matrix, std::ostream &stream) {
43 Matrix3d::Dimensions dim = matrix.dimensions();
44 for (int i = 0; i < dim[2]; ++i) {
45 for (int j = 0; j < dim[0]; ++j) {
46 for (int k = 0; k < dim[1]; ++k) {
47 stream << matrix(j, k, i) << " ";
48 }
49 stream << std::endl;
50 }
51 if (i != (dim[2] - 1)) {
52 stream << "===========================" << std::endl;
53 }
54 }
55}

72inline Matrix3d Matrix3dVectorMinimum(const std::vector<Matrix3d> &matrices) {
73 if (matrices.empty()) {
74 return {};
75 }
76 Matrix3d smallest = matrices[0];
77 auto dims = smallest.dimensions();
78 for (Matrix3d matrix : matrices) {
79 if (matrix.dimensions() != dims) {
80 // throw error
81 return {};
82 }
83 smallest = smallest.cwiseMin(matrix);
84 }
85 return smallest;
86}

131inline double TimedMatrix3dSummedOverDimensions(const TimedMatrix3d &data) {
132 if (data.size() <= 0) {
133 return 0.0;
134 }
135
136 Matrix3d running_total(data.begin()->second.dimensions());
137 running_total = running_total.setZero();
138 for (auto kv : data) {
139 running_total += kv.second;
140 }
141
142 Eigen::Tensor<double, 0> result = running_total.sum();
143 return result(0);
144}

150inline TimedMatrix3d MultiplyTimedMatrix3dByDouble(const TimedMatrix3d &data,
151 const double &value) {
152 TimedMatrix3d result;
153
154 for (int t = 0; t < data.size(); t++) {
155 auto ref = data.at(t);
156 Matrix3d temp =
157 CreateMatrix3d(ref.dimension(0), ref.dimension(1), ref.dimension(2))
158 .constant(value);
159 result[t] = ref * temp;
160 }
161 return result;
162}

168inline TimedMatrix3d MultiplyTimedMatrix3dByMatrix(const TimedMatrix3d &state,
169 const Matrix3d &value) {
170 TimedMatrix3d result;
171 for (int t = 0; t < state.size(); ++t) {
172 Matrix3d temp = state.at(t) * value;
173 result[t] = temp;
174 }
175 return result;
